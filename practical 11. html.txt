    <!-- Web Storage -->
    <div class="card">
      <h2>3. Web Storage API</h2>
      <input type="text" id="username" placeholder="Enter your name" />
      <button onclick="saveData()">Save Name</button>
      <button onclick="showData()">Show Saved Name</button>
      <p id="storageResult">Saved data will appear here.</p>
    </div>

  <script>
    // Drag & Drop
    function allowDrop(event) {
      event.preventDefault();
    }

    function drag(event) {
      event.dataTransfer.setData("text", event.target.id);
      event.target.id = "draggedItem";
    }

    function drop(event) {
      event.preventDefault();
      const data = event.dataTransfer.getData("text");
      const item = document.getElementById("draggedItem");
      event.target.innerHTML = "";
      event.target.appendChild(item);
    }

    // Geolocation
    function getLocation() {
      const output = document.getElementById("locationResult");

      if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(
          function(position) {
            output.innerHTML =
              "Latitude: " + position.coords.latitude +
              "<br>Longitude: " + position.coords.longitude;
          },
          function() {
            output.innerHTML = "Unable to retrieve location.";
          }
        );
      } else {
        output.innerHTML = "Geolocation is not supported by this browser.";
      }
    }

    // Web Storage
    function saveData() {
      const name = document.getElementById("username").value;
      localStorage.setItem("savedName", name);
      document.getElementById("storageResult").innerHTML = "Name saved successfully!";
    }

    function showData() {
      const savedName = localStorage.getItem("savedName");
      document.getElementById("storageResult").innerHTML =
        savedName ? "Saved Name: " + savedName : "No data found.";
    }
  </script>
